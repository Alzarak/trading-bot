    ```python
    class AuditLogger:
        """NDJSON audit logger for Claude trade decisions.

        Every Claude recommendation and its execution outcome is written
        as a single JSON line to {data_dir}/audit/claude_decisions.ndjson.
        """

        def __init__(self, data_dir: Path | None = None):
            data_dir = data_dir or Path(os.environ.get("CLAUDE_PLUGIN_DATA", "."))
            self.audit_dir = data_dir / "audit"
            self.audit_dir.mkdir(parents=True, exist_ok=True)
            self.audit_file = self.audit_dir / "claude_decisions.ndjson"
            self.session_id = datetime.now(timezone.utc).strftime("%Y%m%dT%H%M%S")

        def log_recommendation(self, rec: ClaudeRecommendation) -> None:
            """Write recommendation to audit log as NDJSON line."""
            entry = {
                "timestamp": datetime.now(timezone.utc).isoformat(),
                "session_id": self.session_id,
                "type": "recommendation",
                "symbol": rec.symbol,
                "action": rec.action,
                "confidence": rec.confidence,
                "reasoning": rec.reasoning,
                "strategy": rec.strategy,
                "atr": rec.atr,
                "stop_price": rec.stop_price,
            }
            with open(self.audit_file, "a") as f:
                f.write(json.dumps(entry) + "\n")
            logger.info("Audit: logged recommendation for {} — {} (confidence={:.2f})",
                       rec.symbol, rec.action, rec.confidence)

        def log_execution_result(self, rec: ClaudeRecommendation, status: str, order_id: str | None = None, reason: str = "") -> None:
            """Write execution outcome alongside the recommendation."""
            entry = {
                "timestamp": datetime.now(timezone.utc).isoformat(),
                "session_id": self.session_id,
                "type": "execution",
                "symbol": rec.symbol,
                "action": rec.action,
                "status": status,  # "submitted", "blocked", "failed"
                "order_id": order_id,
                "reason": reason,
                "confidence": rec.confidence,
                "reasoning": rec.reasoning,
            }
            with open(self.audit_file, "a") as f:
                f.write(json.dumps(entry) + "\n")

        def get_session_decisions(self) -> list[dict]:
            """Return all decisions from current session."""
            if not self.audit_file.exists():
                return []
            decisions = []
            for line in self.audit_file.read_text().splitlines():
                try:
                    entry = json.loads(line)
                    if entry.get("session_id") == self.session_id:
                        decisions.append(entry)
                except json.JSONDecodeError:
                    continue
            return decisions
    ```

          ```python
          def get_analysis_context(scanner: MarketScanner, config: dict) -> dict:
              """Generate analysis context for all watchlist symbols.

              Returns dict of {symbol: {"df_summary": str, "prompt": str, "current_price": float}}
              for the /run command agent mode to use.
              """
              analyzer = ClaudeAnalyzer(config)
              watchlist = config.get("watchlist", [])
              context = {}
              for symbol in watchlist:
                  df = scanner.scan(symbol)
                  if df.empty:
                      continue
                  current_price = float(df.iloc[-1]["close"])
                  for strategy_config in config.get("strategies", []):
                      strategy_name = strategy_config.get("name", "unknown")
                      prompt = analyzer.build_analysis_prompt(symbol, df, strategy_name)
                      key = f"{symbol}_{strategy_name}"
                      context[key] = {
                          "symbol": symbol,
                          "strategy": strategy_name,
                          "prompt": prompt,
                          "current_price": current_price,
                      }
              return context
          ```

       d) Add an `execute_claude_recommendation()` function:
          ```python
          def execute_claude_recommendation(
              recommendation_json: str,
              executor: OrderExecutor,
              tracker: PortfolioTracker,
              state_store: StateStore,
              audit_logger: AuditLogger,
              analyzer: ClaudeAnalyzer,
          ) -> dict:
              """Parse a Claude recommendation and execute through risk manager.

              Returns execution result dict with status.
              """
              recs = analyzer.parse_response(recommendation_json)
              results = []
              for rec in recs:
                  audit_logger.log_recommendation(rec)
                  signal = rec.to_signal()
                  if signal.action in ("BUY", "SELL"):
                      # current_price must come from context
                      # For now, use stop_price as proxy or caller provides
                      order = executor.execute_signal(signal, rec.stop_price)
                      status = "submitted" if order is not None else "blocked"
                      order_id = str(getattr(order, "id", None)) if order else None
                      audit_logger.log_execution_result(rec, status, order_id)
                      results.append({"symbol": rec.symbol, "action": rec.action, "status": status})
                  else:
                      audit_logger.log_execution_result(rec, "hold", reason="HOLD signal")
                      results.append({"symbol": rec.symbol, "action": "HOLD", "status": "hold"})
              return {"results": results}
          ```
